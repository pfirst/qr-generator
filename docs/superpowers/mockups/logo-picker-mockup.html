Implementation — additive ล้วน (design v2: ง่ายกว่าเดิมมาก)

โลโก้เป็น badge สีพร้อมใช้ → ป้อนเข้า style.logo (data URL) ตรงๆ เหมือน path อัปโหลดเดิม ไม่มี compose / contrast-guard / brand-color background อีกแล้ว pipeline เดิมทำงานต่อไม่ต้องแก้

เลือก preset ─▶ style.logo = "data:image/png;base64,…" (หรือ svg ของ Thai QR Payment) style.logoPreset = 'bank:SCB' | 'thaiqr' | 'pp:TrueMoney' │ renderQrSvg(payload, ecc='H', style, …) ← เดิม ไม่แก้ │ preview · export.ts · verify.ts · HistoryCard ← เดิม ไม่แก้

logoPresets.ts — โครงหลัก (สั้นลงเยอะ)

// PNG → data URL ตอน build (Vite ?inline ฝัง base64 ในบันเดิล, ไม่มี network ตอนรัน)
const pngs = import.meta.glob('../assets/logos/banks/*.png', { query:'?inline', eager:true, import:'default' })
import meta from '../assets/logos/banks-logo.json'
import thaiqrRaw from '../assets/logos/thaiqr.svg?raw'

export interface LogoPreset { id:string; name:string; nameEN:string; url:string }

const url = (sym:string) => pngs[`../assets/logos/banks/${sym}.png`] as string

// หมวดธนาคาร (ตัด PromptPay/TrueMoney ออก) เรียงแบงค์ใหญ่ก่อน
const BILL_ORDER = ['SCB','KBANK','KTB','BBL','BAY','TTB','GSB','GHB','BAAC',
                    'KKP','TISCO','TCRB','LHB','IBANK','UOB','CIMB']  // แบงค์ที่คนใช้บ่อย (ตัด CITI/HSBC/ICBC)
export const BANK_PRESETS: LogoPreset[] = BILL_ORDER.map(s => ({ id:s, ...meta[s], url:url(s) }))

// หมวด PromptPay: Thai QR Payment (svg ของพี่เฟิส) + PromptPay + TrueMoney
const svgDataUrl = (svg:string) => 'data:image/svg+xml;base64,' + btoa(unescape(encodeURIComponent(svg)))
export const PP_PRESETS: LogoPreset[] = [
  { id:'thaiqr', name:'Thai QR Payment', nameEN:'Thai QR Payment', url: svgDataUrl(squareWrap(thaiqrRaw)) },
  { id:'PromptPay', ...meta.PromptPay, url:url('PromptPay') },
  { id:'TrueMoney', ...meta.TrueMoney, url:url('TrueMoney') },
]

การ wire เข้า state

  • เลือก preset → patchStyle({ logo: preset.url, logoPreset: 'bank:'+id })
  • อัปโหลดเอง → logo = <data url>, logoPreset = null
  • ลบ → logo:null, logoPreset:null
  • ECC: App มี effEcc = style.logo ? 'H' : style.ecc อยู่แล้ว → set logo อย่างเดียวก็บังคับ H

หมายเหตุ / trade-off

  • PNG 200×200 เป็น raster — export ใหญ่ (1024px, โลโก้ ~32% ≈ 328px) จะ upscale เล็กน้อย ยอมรับได้ ยกเว้นพิมพ์ใหญ่มาก (Thai QR Payment เป็น SVG คมเสมอ)
  • โลโก้ส่วนใหญ่มีวงกลมสีแบรนด์ในตัว → คอนทราสต์ดีอยู่แล้ว วางตรงๆ ได้ (hideBackgroundDots:true เคลียร์ dot ใต้โลโก้ให้สะอาด)
  • decode re-check (verify.ts) ทำงานเหมือนเดิม โลโก้ใหญ่ไปจะเตือน "สแกนไม่ติด" อัตโนมัติ
  • ตัด CITI/HSBC/ICBC ออกแล้ว (ไม่มี retail ในไทย) เหลือ 16 แบงค์ที่คนใช้บ่อย — เพิ่มกลับได้ที่ BILL_ORDER
